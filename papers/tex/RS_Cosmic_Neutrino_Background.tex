\begin{abstract}
The Cosmic Neutrino Background (C$\nu$B) is the relic background of
neutrinos from the early universe, predicted to have a temperature
$T_\nu = (4/11)^{1/3} T_\gamma \approx 1.946$ K and a number density
$n_\nu \approx 336$ cm$^{-3}$ per flavor.  In Recognition Science,
neutrinos are fermionic (4-tick, half-integer spin) ledger modes carrying
the three Dirac neutrino masses at deep rungs $r_\nu \in \{-54, -58, -63\}$
on the $\varphi$-ladder (Lean: \texttt{Physics.NeutrinoSector}, proved).
The C$\nu$B temperature follows from the RS-derived CMB temperature
$T_0 = 2.725$ K (\texttt{RS\_CMB\_Temperature.tex}) combined with the
$(4/11)^{1/3}$ cooling factor from $e^+e^-$ annihilation.  The neutrino
number density uses the Fermi-Dirac statistics for massless (or
ultra-relativistic) fermions (Lean: \texttt{Thermodynamics.FermiDirac}).
The C$\nu$B contributes $N_{\rm eff} = 3.046$ effective neutrino species
to the radiation density.
\end{abstract}

\section{Introduction}

The C$\nu$B is the neutrino analogue of the CMB.  Neutrinos decouple
at $T \sim 2$ MeV (weak interaction freeze-out), then $e^+e^-$
annihilation heats the photon bath but not the neutrinos, giving
$T_\nu/T_\gamma = (4/11)^{1/3}$.

\section{Derivation of C$\nu$B Temperature}

\subsection{Decoupling Temperature}

Neutrinos decouple when $\Gamma_{\rm weak} \sim H$:
\begin{equation}
  T_{\rm dec} \approx \left(\frac{m_W^4}{M_{\rm Pl}}\right)^{1/3}
  \approx 2\,\text{MeV}.
\end{equation}
With $m_W \approx 80.4$ GeV (RS: \texttt{predict\_mass}) and
$M_{\rm Pl} = c^2\sqrt{\hbar c/G}$ (RS: \texttt{G\_rs\_eq}): $T_{\rm dec} \approx 2$ MeV.

\subsection{$e^+e^-$ Annihilation Heating}

After decoupling, $e^+e^-$ annihilation heats photons:
\begin{equation}
  T_\nu = \left(\frac{4}{11}\right)^{1/3} T_\gamma
  \approx 0.7138\,T_\gamma.
\end{equation}

\begin{theorem}[C$\nu$B temperature]\label{thm:Tnu}
With $T_\gamma = T_0 = 2.72548$ K (Lean: \texttt{Physics.CMB.rs\_cmb\_approx\_2725}):
\begin{equation}
  T_\nu = \left(\frac{4}{11}\right)^{1/3} \times 2.72548\,\text{K}
  = 1.9454\,\text{K}.
\end{equation}
Experimental target: $T_\nu = 1.9454$ K (PTOLEMY experiment~\cite{PTOLEMY2022}).
\end{theorem}

\subsection{Number Density}

The C$\nu$B number density per flavor:
\begin{equation}
  n_\nu = \frac{3}{11}n_\gamma = \frac{3}{11}\frac{2\zeta(3)}{\pi^2}T_\gamma^3
  \approx 56\,\text{cm}^{-3}\text{ per flavor} \times 2\,(\nu+\bar\nu) = 112\,\text{cm}^{-3}.
\end{equation}
Total (3 flavors): $n_{\nu,\rm total} \approx 336$ cm$^{-3}$.

\subsection{Effective Relativistic Degrees of Freedom}

The C$\nu$B contributes to the radiation density:
\begin{equation}
  \rho_{\rm rad} = \rho_\gamma\left[1 + \frac{7}{8}N_{\rm eff}
  \left(\frac{T_\nu}{T_\gamma}\right)^4\right],
  \quad N_{\rm eff} = 3.046.
\end{equation}
The $N_{\rm eff} = 3.046$ (slightly above 3) arises from incomplete
neutrino decoupling and QED corrections.  In RS, this correction uses
the finite-temperature QED structure (\texttt{RS\_Electroweak\_Phase\_Transition.tex}).
