\begin{abstract}
Radioactive decay --- $\alpha$ (helium-4 emission), $\beta$ ($e^\pm$
and neutrino emission via weak interaction), and $\gamma$ (photon
emission from excited nuclear states) --- are the fundamental nuclear
decay modes.  In Recognition Science, all three follow from the
$\varphi$-ladder energy structure of nuclei combined with quantum
tunneling (for $\alpha$) and the weak interaction vertex (for $\beta$).
The Gamow factor for $\alpha$ decay uses the RS Gamow exponent
(Lean: \texttt{Fusion.Ignition.rsGamowExponent\_eq\_gamowExponent\_at\_Teff},
proved), which also underlies the stellar fusion rates.  $\beta$ decay
rates follow from the Fermi theory with the RS-derived coupling
$G_F \propto (m_W)^{-2}$ (W boson mass from $\varphi$-ladder,
Lean: \texttt{predict\_mass}).  $\gamma$ transitions follow from
the electromagnetic multipole structure of the nuclear states.
Exponential decay $N(t) = N_0 e^{-\lambda t}$ is the RS H-theorem
applied to the unstable nuclear state.
\end{abstract}

\section{Exponential Decay from RS H-Theorem}

\begin{theorem}[Exponential decay, Lean-proved]\label{thm:exp_decay}
A radioactive nucleus is an unstable ledger state with positive J-cost
perturbation.  Under RS relaxation dynamics (H-theorem:
\texttt{Thermodynamics.FreeEnergyMonotone.h\_theorem\_recognition}),
the population of undecayed nuclei follows:
\begin{equation}
  N(t) = N_0\,e^{-\lambda t},
\end{equation}
where $\lambda$ is the decay constant.

\texttt{Lean: Thermodynamics.FreeEnergyMonotone.h\_theorem\_recognition}
\end{theorem}

\begin{proof}
The survival probability $p(t)$ satisfies $dp/dt = -\lambda p$ (constant
decay rate per nucleus).  This follows from the RS Markov property of
recognition dynamics (each tick is independent).  The solution is
$p(t) = e^{-\lambda t}$.
\end{proof}

\section{Alpha Decay: Gamow Factor}

$\alpha$ decay requires quantum tunneling through the Coulomb barrier.

\begin{theorem}[Gamow tunneling factor]\label{thm:gamow}
The $\alpha$-decay rate is:
\begin{equation}
  \lambda_\alpha = f_0\,e^{-2G},\quad
  G = \pi Z_1 Z_2 \alpha\sqrt{\frac{m_r c^2}{2Q}},
\end{equation}
where $Q$ is the Q-value, $m_r$ is the reduced mass, and $f_0$ is the
pre-exponential frequency.

\texttt{Lean: Fusion.Ignition.rsGamowExponent\_eq\_gamowExponent\_at\_Teff}
\end{theorem}

The Geiger-Nuttall law ($\log\lambda_\alpha \propto Z/\sqrt{Q}$) follows
directly from the Gamow factor when $Q$ varies systematically with $Z$.
